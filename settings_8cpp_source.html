72status_t Settings::StartMonitoring() {
73 if (! fTarget) { return B_BAD_HANDLER; }
74 if (fMonitoringActive) { return B_OK; }
75
76 BPath* pathToSettingsFile = GetPathToSettingsFile();
77 if (! pathToSettingsFile) { return B_BAD_VALUE; }
78
79 // ---==< Entering critical section >==---
80 fLock.Lock();
81
82 // Find the settings file
83 BEntry entry(pathToSettingsFile.Path(), true);
84 if (!entry.Exists()) {
85 // There's no settings file
86 if ( B_OK != CreateSettingsFile()) {
87 fLock.Unlock();
88 return B_ENTRY_NOT_FOUND;
89 }
90 }
91
92 // Get the node reference
93 node_ref nref;
94 status_t status = entry.GetNodeRef(&nref);
95 if (B_OK != status) { fLock.Unlock(); return status; }
96
97 // Start monitoring
98 status = watch_node(&fNodeRef, B_WATCH_STAT, target);
99 if (status == B_OK) {
100 fMonitoringActive = true;
101 }
102
103 fLock.Unlock();
104 // ---==< Exitting critical section >==---
105
106 return status;
107}

119status_t Settings::CreateSettingsFile() {
120 BInputDevice* device = NULL;
121 BMessage toSave = new BMessage('CONF');
122 BPath* pathToSettingsFile = GetPathToSettingsFile();
123 if (! pathToSettingsFile) { return B_BAD_VALUE; }
124
125 // Initialize the settings file
126 BFile file(fPath.Path(), B_WRITE_ONLY | B_CREATE_FILE);
127 status_t status = file.InitCheck();
128 if (status != B_OK) { return status; }
129 file.Lock(); // <==== LOCK THE FILE FROM ALL OTHER ACCESSES!
130
131 // Obtain the list of all devices
132 BList devices;
133 status = get_input_devices(&devices);
134 if (B_OK != status) { file.Unlock(); return status; }
135
136 // Populate the BMessage toSave with pointing devices
137 int32 listSize = devices.CountItems();
138 int32 i = 0;
139 while (i < listSize) {
140 device = devices.ItemAt(i);
141 if (!device ||
142 device->Type() != B_POINTING_DEVICE)
143 { i++; continue; }
144
145 BMessage singleDevice(i); // The "what" field will be the number of device
146 singleDevice.AddString("name", device->Name());
147 singleDevice.AddBool("ignored", false);
148 toSave->AddMessage(&singleDevice);
149 }
150
151 // Dispose of the device list.
152 list_o_devices.DoForEach (del_InputDevice);
153 list_o_devices.MakeEmpty();
154
155 // Save the BMessage with settings
156 status = toSave.Flatten(&file);
157 file.Unlock(); // <==== UNLOCK THE FILE, enable access
158 if (status != B_OK) { return status; }
159
160 printf("[SettingsMonitor] Created new settings file: %s\n", fPath.Path());
161 return B_OK;
162}
